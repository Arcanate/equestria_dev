country_event = { #The Reformisten are approached by the Empire to be puppeted once more
	id = katerin_new.9
	immediate = {log = "[GetDateText]: [Root.GetName]: event katerin_new.9"}
	title = katerin_new.9.t
	desc = katerin_new.9.d
	picture = GFX_event_griffon_back
	
	is_triggered_only = yes
	#fire_only_once = yes
	
	option  ={ #accept
		name = katerin_new.9.a
		trigger = {
			is_puppet = no
		}		
		ai_chance = {
			factor = 5 #Eh I'll give em a small chance to accept anyway, 1 in 20
			modifier = {
				is_puppet = yes #Redundant check is redundant but eh, just in case.
				factor = -5
			}
		}
		KAT = { puppet = ROOT }
	}
	option = { #refuse
		name = katerin_new.9.b
		ai_chance = {
			factor = 95 #Why would knightly orders even want to be puppeted by commies?
		}
		add_political_power = -25
		KAT = { country_event = { id = katerin_new.10 days =1 } }
	}
}
