	is_triggered_only = yes

		name = kasa.20.a
		hidden_effect = {
			find_highest_in_array = {
				array = ZES_election_array
				index = ZES_election_max
			}
		    if = {
			limit = { check_variable= { ZES_election_max = 0 } }
			country_event = { id = kasa.21 days = 1 }
		    }
		    if = {
			limit = { check_variable= { ZES_election_max = 1 } }
			country_event = { id = kasa.22 days = 1 }
		    }
		    if = {
			limit = { check_variable= { ZES_election_max = 2 } }
			country_event = { id = kasa.23 days = 1 }
		    }
		    if = {
			limit = { check_variable= { ZES_election_max = 3 } }
			country_event = { id = kasa.24 days = 1 }
		    }
		    clear_array = ZES_election_array
		}
	}
}
