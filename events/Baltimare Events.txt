			var = BAL_libsoc_influence
			value = 50
		}

			value = 50
		}
		set_variable = {
			var = BAL_min_influence
			value = 0
		}
		set_variable = {
			var = BAL_max_influence
			value = 100
		}
		clamp_variable = {
			var = BAL_libsoc_influence
			min = BAL_min_influence
			max = BAL_max_influence
		}
		clamp_variable = {
			var = BAL_authsoc_influence
			min = BAL_min_influence
			max = BAL_max_influence
		}
		hidden_effect = {
			if = {
				limit = {
					SIC = {
						exists = yes
					}
				}
				SIC = {
					country_event = { id = baltimare.3 days = 3 }
				}
			}
		}
	}
}
