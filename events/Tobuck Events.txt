country_event = { ### Focus The Art of the Deal
	id = tobuck_start.1 
	title = tobuck_start.1.t
	desc = tobuck_start.1.d
	is_triggered_only = yes
	option = {
		name = tobuck_start.1.a
		custom_effect_tooltip = TBK_explain_early_gameplay_mechanics
		hidden_effect = {
			set_variable = {
				var = TBK_guns_pile
				value = 100
			}
			clamp_variable = {
	            var = TBK_guns_pile
	            min = 0
	            max = 100
            }
			set_variable = {
				var = TBK_dirty_money
				value = 15
			}
			set_variable = {
				var = TBK_pea_vault
				value = 0
			}
			set_variable = {
				var = TBK_deal_reveal_timer_var
				value = 75
			}
			set_variable = {
				var = TBK_5_guns_cost
				value = 25
			}
			set_variable = {
				var = TBK_10_guns_cost
				value = 50
			}
			set_country_flag = TBK_is_hiding_deal
			activate_mission = TBK_deal_reveal_mission
		}
	}
}

	id = tobuck_start.15
	title = tobuck_start.15.t
	desc = tobuck_start.15.d
	is_triggered_only = yes

		if = {
			limit = {

					TBK_pea_vault < 19
				}
			}
			country_event = { id = tobuck_start.17 }
		}
		if = {
			limit = {
				check_variable = {
					TBK_pea_vault > 19

				}
			}
			country_event = { id = tobuck_start.18 }
		}
		if = {
			limit = {
				check_variable = {
					TBK_pea_vault > 39
				}
				check_variable = {
					TBK_pea_vault < 70
				}
			}
			country_event = { id = tobuck_start.19 }
		}
		if = {
			limit = {
				check_variable = {
					TBK_pea_vault > 69
				}
				check_variable = {
					TBK_pea_vault < 99
				}
			}
			country_event = { id = tobuck_start.20 }
		}

		}
	}
