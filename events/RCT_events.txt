country_event = {
	id = rivercity.7
	immediate = {log = "[Root.GetName]: event rivercity.7"}#Auto-logging
	title = rivercity.7.t
	desc = rivercity.7.d
	picture = GFX_event_lake_city

	is_triggered_only = yes

	option = {
		name = rivercity.7.a
		if = {
			limit = {
				ROOT = {
            		check_variable = { RCT_Unrest > 6.0 }
				}
			}
			ROOT = { complete_national_focus = RCT_Socialist_Agitation }
			}
		else = { ROOT = { complete_national_focus = RCT_The_Mandate_Continues } }
	}
}
