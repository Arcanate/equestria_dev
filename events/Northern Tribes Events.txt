country_event = {	#THE UNCLE; Fiana talks to Oskar about modernism
	id = NTR_fiana.2
	title = NTR_fiana.2.t
	desc = NTR_fiana.2.d

	is_triggered_only = yes

	option = {	#Fiana rejects modernism
		name = NTR_fiana.2.a
		add_political_power = 50
	}

	option =  {	#Fiana embraces modernism
		trigger = {
			NOT = {
				has_country_flag = NTR_embraced_tradition
			}
		}
		name = NTR_fiana.2.b
		set_country_flag = NTR_embraced_modernism
		swap_ideas = {
			remove_idea = NTR_the_untamed_north_1
			add_idea = NTR_the_untamed_north_2
		}
		custom_effect_tooltip = NTR_embrace_modernism_tt
	}
	
}
