### Improving a shithole ###
country_event = {
	id = stagpal.5
	immediate = {log = "[GetDateText]: [Root.GetName]: event stagpal.5"}#Auto-logging
	title = stagpal.5.t
	desc = stagpal.5.d
		
	#picture = GFX_event_lake_city

	is_triggered_only = yes

	option = { #great
		name = stagpal.5.a
		random_owned_controlled_state = {
			limit = {
				free_building_slots = {
					building = infrastructure
					size > 0
					include_locked = yes
				}
				OR = {
					is_in_home_area = yes
					NOT = {
						owner = {
							any_owned_state = {
								free_building_slots = {
									building = infrastructure
									size > 0
									include_locked = yes
								}
								is_in_home_area = yes
							}
						}
					}
				}
			}
			add_building_construction = {
				type = infrastructure
				level = 1
				instant_build = yes
			}
		}
	}
}
