		name = bakara.1.a
		custom_effect_tooltip = BAK_president_tp
		remove_ideas = BAK_elections

		}
		add_popularity = { 
		    ideology = fascism 
			popularity = 0.10
		}
		hidden_effect = {
		    load_focus_tree = { 
			    tree = bakara_focus_fascists
				keep_completed = yes 
			}
		    if = {
			    limit = {
				    check_variable = {
                        var = BAK_electoral_unrest
                        value = 70
                        compare = greater_than_or_equals
                    }
				}
				country_event = { 
	                id = bakara.17
	                days = 7
                }
			}
			else = {
				country_event = { 
	                id = bakara.21
	                days = 30
                }
			}
		}
	}
}
