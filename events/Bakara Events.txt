country_event = { #BFZ Wins The Election
	id = bakara.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event bakara.1"}
	title = bakara.1.t
	desc = bakara.1.d
	picture = GFX_event_river_city

	is_triggered_only = yes

	option = {
		name = bakara.1.a
		custom_effect_tooltip = BAK_president_tp
		remove_ideas = BAK_elections
		set_politics = {
			ruling_party = fascism
			elections_allowed = no
		}
		add_popularity = { 
		    ideology = fascism 
			popularity = 0.10
		}
		hidden_effect = {
		    complete_national_focus = BAK_election
		    if = {
			    limit = {
				    check_variable = {
                        var = BAK_electoral_unrest
                        value = 70
                        compare = greater_than_or_equals
                    }
				}
				country_event = { 
	                id = bakara.17
	                days = 7
                }
			}
			else = {
				country_event = { 
	                id = bakara.21
	                days = 30
                }
			}
		}
	}
}
