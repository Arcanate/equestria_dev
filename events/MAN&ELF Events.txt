add_namespace = MAN&ELF

country_event = { 
	id = MAN&ELF.0 ## the ELF declaration of insurgency ##
	immediate = {log = "[GetDateText]: [Root.GetName]: event MAN&ELF.0"}#Auto-logging
	title = MAN&ELF.0.t
	desc = MAN&ELF.0.d

	picture = GFX_event_griffon_back
	is_triggered_only = yes
	trigger = {
		MAN = {
			is_puppet_of = CHN 
			NOT = { has_war_with = ELF }
		}
	}
	option = {
		name = MAN&ELF.0.a
		add_stability = -0.60
		start_civil_war = {
			ruling_party = fascism
			ideology = democratic
			size = 0
			only_own_territory = yes 
		}
		every_country = {
			limit = { 
				has_war_with = MAN
				original_tag = MAN
				civilwar_target = MAN
			}
			save_event_target_as = the_placeholder_for_ELF_to_go_to_war_with_MAN
		}
	}
}
