	id = bufstate.34
	immediate = {log = "[GetDateText]: [Root.GetName]: event bufstate.34"}
	immediate = {

	}
	title = bufstate.34.t
	desc = bufstate.34.d
	picture = GFX_event_canyons

	is_triggered_only = yes


	option = {
		name = bufstate.34.a
		add_to_array = { #thunderhoof popularity
			array = BUF_power_struggle_array
			value = party_popularity@neutrality
			index = 0
		}
		add_to_array = { #farwander popularity
			array = BUF_power_struggle_array
			value = party_popularity@democratic
			index = 0
		}
		add_to_array = { #greyhorn popularity
			array = BUF_power_struggle_array
			value = party_popularity@fascism
			index = 0
		}
		find_highest_in_array = {
			array = BUF_power_struggle_array
			index = BUF_power_struggle_max
		}
		set_variable = { test = BUF_power_struggle_max }
		if = {
			limit = { check_variable = { BUF_power_struggle_max = 0 } }
			custom_effect_tooltip = BUF_thunderhoof_wins_tt
			#thunderhoof wins
		}
		else_if = {
			limit = { check_variable = { BUF_power_struggle_max = 1 } }
			custom_effect_tooltip = BUF_farwander_wins_tt
			#farwander wins
		}
		else = {
			custom_effect_tooltip = BUF_greyhorn_wins_tt
			#greyhorn wins
		}
		clear_array = BUF_power_struggle_array
	}
}
