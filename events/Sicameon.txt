country_event = { ## Elections!
	id = sicameon.15
	title = sicameon.15.t
	desc = sicameon.15.d
	picture = GFX_event_griffon_mountain ##placeholder
	
	is_triggered_only = yes
	
	fire_only_once = yes
	option = { #Democrats win!
		trigger = {
			check_variable = { party_popularity@democratic > party_popularity@communism }
		}
		name = sicameon.15.a
		set_country_flag = SIC_harmonic_victory
	}
	
	option = { #Syndies win!
		trigger = {
			check_variable = { party_popularity@communism > party_popularity@democratic }
		}
		name = sicameon.15.b
		set_country_flag = SIC_communist_victory
	}
	
	option = { #fascists rig the elections
		trigger = {
			fascism > 0.35
		}
		name = sicameon.15.c
		set_country_flag = SIC_fascist_victory
	}
