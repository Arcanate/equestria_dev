### Black Claw Bandits
country_event = {
	id = mareland.5
	title = mareland.5.t
	desc = mareland.5.d
	picture = GFX_event_griffon_with_gun
	
	fire_only_once = yes
	
	trigger = {
		tag = EQC
		has_completed_focus = EQC_force_griffon_conscription
		any_owned_state = {
			industrial_complex > 0
		}
	}
	
	mean_time_to_happen = {
		months = 3
	}
	
	option = {
		name = mareland.5.a
		hidden_effect = { set_country_flag = { flag = EQC_black_claw_bandit_raids value = 1 } }
			
		random_owned_state = {
			limit = {
				industrial_complex > 0
			}
			damage_building = {
				type = industrial_complex
				damage = 0.7
			}
		}
		
		custom_effect_tooltip = EQC_bandit_raids_start_tt
	}
}
