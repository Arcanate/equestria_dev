	id = mareland.1
	title = mareland.1.t
	desc = mareland.1.d
	picture = GFX_event_ponies_in_battle

		tag = EQC
		has_war_with = CHN
		NOT = {
			any_other_country = {
				limit = {
					is_on_griffon_continent = yes
					has_war_with = EQC
				}
			}
		}
		
		EQS = {
			has_war_with = CHN
			has_capitulated = yes
		}
	}

	option = {
		name = mareland.1.a
		CHN = {
			country_event = { id = mareland.2 hours = 2 }
		}
	}

		name = mareland.1.b
	}
}

		}

		add_manpower = 5000
		
	}
	option = { # Twilight
		name = mareland.4.c
		add_tech_bonus = { bonus = 0.5 uses = 2 category = race_tech }
	}
	option = { # Cadance

			NOT = {
				CRY = {
					OR = {
						has_country_flag = royalty_executed

					}
				}
			}
		}
		
		add_political_power = 75
		
		name = mareland.4.e
	}
}
