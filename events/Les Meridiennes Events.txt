	id = meridiennes.13
	immediate = { log = "[GetDateText]: [Root.GetName]: event meridiennes.13" }
	title = meridiennes.13.t
	desc = meridiennes.13.d
	
	fire_only_once = yes
	is_triggered_only = yes
	picture = GFX_event_griffon_soldier
	option = {
		name = meridiennes.13.a
		MER = { get_current_government_type = yes }
				JER = {
					puppet = MER
				if = {
					limit = {
						AND = {
							has_dlc = "Together for Victory" 
							has_dlc = "Death or Dishonor"
						}
					}
					set_autonomy = {
						target = MER
						autonomy_state = autonomy_colony
					}
				}
			}
			MER = { restore_previous_government_type = yes }		
			load_focus_tree = {
				tree = MER_loyalist_tree
				keep_completed = yes
			}
			complete_national_focus = MER_steady_as_she_goes
	}
}
