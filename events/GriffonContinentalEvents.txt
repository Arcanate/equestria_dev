country_event = { ## Set proper science base
    id = griffonian.993
    title = griffonian.993.t
    desc = griffonian.993.d
	picture = GFX_event_griffon_mountain

		trigger = {
			has_poor_science_ideas = yes
			OR = {
				AND = {
					owns_major_griffon_city = yes
					OR = {
						num_of_factories > 17
						is_major_griffon_core = yes
					}
				}
				any_owned_state = {
					AND = {
						is_on_continent = europe
						state_population > 200000
					}
				}
				any_country = {
					AND = {
						is_subject_of = ROOT
						is_on_equestrian_continent = yes
					}
				}
				has_country_flag = proper_science_base_unclocked
			}
		}
		mean_time_to_happen = { days = 60 }
    option = {
    	name = griffonian.993.a
	    remove_ideas = big_science_base
			remove_ideas = better_science_base
			remove_ideas = poor_science_base
		}

}

country_event = { ## Set big science base
    id = griffonian.992
    title = griffonian.992.t
    desc = griffonian.992.d
	picture = GFX_event_griffon_mountain

		trigger = {
			OR = {
				AND = {
					has_idea = better_science_base
					OR = {
						AND = {
							owns_big_griffon_city = yes
							OR = {
								num_of_factories > 9
								hidden_trigger = {
									is_big_griffon_core = yes
								}
							}
						}
						has_country_flag = big_science_base_unlocked
					}
				}
				AND = {
					has_idea = big_science_base
					NOT = {
						OR = {
							AND = {
								owns_big_griffon_city = yes
								OR = {
									num_of_factories > 9
									hidden_trigger = {
										is_big_griffon_core = yes
									}
								}
							}
							has_country_flag = big_science_base_unlocked
						}
					}
				}
			}
		}
		mean_time_to_happen = {
			days = 30
			modifier = {
				factor = 6
				has_idea = big_science_base
			}
		}
    option = {
    	name = griffonian.993.a
			trigger = {
				has_idea = better_science_base
			}
			swap_ideas = { add_idea = big_science_base remove_idea = better_science_base }
		}
		option = {
    	name = griffonian.993.b
			trigger = {
				has_idea = big_science_base
			}
			swap_ideas = { remove_idea = big_science_base add_idea = better_science_base }
		}
}

country_event = { ## Set better science base
    id = griffonian.991
    title = griffonian.991.t
    desc = griffonian.991.d
	picture = GFX_event_griffon_mountain

			OR = {
