country_event = { #Change of govt generic griffon tree fascism
	id = griffonian.3
	title = griffonian.3.t
	desc = {
		text = griffonian.3.d_change
		trigger = { NOT = { has_government = fascism } }
	}
	desc = {
		text = griffonian.3.d
		trigger = { has_government = fascism }
	}
	picture = GFX_event_griffon_mountain
	is_triggered_only = yes
	option = {
		name = griffonian.3.a
		trigger = { NOT = { has_government = fascism } }
		add_political_power = -150
		set_politics = { ruling_party = fascism elections_allowed = no }
	}
	option = {
		name = griffonian.3.b
		trigger = { has_government = fascism }
		add_political_power = 100
	}
}

country_event = { #Cosmetic tag for minors
	id = griffonian.1000
	title = griffonian.1000.t
	desc = griffonian.1000.d
	picture = GFX_event_griffon_mountain

    trigger = {
        is_griffon_continent_country = yes
				is_griffon_race = yes
        NOT = {
        	original_tag = GRI
        }
        owns_state = 382
        NOT = { has_country_flag = holds_griffon_capital }
		NOT = { country_exists = GRI }
    }
    mean_time_to_happen = { days = 1 }
	option = {
	name = griffonian.1000.a
	if = { limit = { has_government = fascism
	 			}
	 	set_cosmetic_tag = GRIF_EMPIRE
	 	set_country_flag = holds_griffon_capital
			}
	if = { limit = { has_government = communism
	 			}
	 	set_cosmetic_tag = GRIF_COM
	 	set_country_flag = holds_griffon_capital
			}
	if = { limit = { has_government = neutrality
	 			}
	 	set_cosmetic_tag = GRIF_EMPIRE
	 	set_country_flag = holds_griffon_capital
			}
	if = { limit = { has_government = democratic
	 			}
	 	set_cosmetic_tag = GRIF_EMPDEM
	 	set_country_flag = holds_griffon_capital
			}
		}
	}
