add_namespace = river_cup

news_event = { #River cup 1007
	id = river_cup.1
	immediate = {log = "[Root.GetName]: event river_cup.1"}#Auto-logging
	title = river_cup.1.t
	desc = river_cup.1.d
	picture = GFX_news_event_friendship_games

	is_triggered_only = yes

	trigger = {
		is_on_griffon_continent = yes
		NOT = {
			any_country = {
				has_idea = RIV_member_state
				has_war = yes
			}
		}
	}

	immediate = {
		hidden_effect = {
			set_global_flag = rc_calculated
			river_cup_calculate = yes
		}
	}

	option = {
		name = river_cup.1.a
		trigger = {
			owns_state = var:global.global.rc_winner
		}
		hidden_effect = {
			clr_global_flag = rc_calculated
		}
		add_political_power = 10
	}

	option = {
		name = river_cup.1.b
		trigger = {
			owns_state = var:global.global.rc_runner_up
		}
		hidden_effect = {
			clr_global_flag = rc_calculated
		}
		add_political_power = 5
	}

	option = {
		name = river_cup.1.c
		trigger = {
			has_idea = RIV_member_state
			NOT = {
				owns_state = var:global.global.rc_winner
				owns_state = var:global.global.rc_runner_up
			}
		}
		hidden_effect = {
			clr_global_flag = rc_calculated
		}
	}

	option = {
		name = river_cup.1.e
		trigger = {
			NOT = {
				has_idea = RIV_member_state
			}
		}
		hidden_effect = {
			clr_global_flag = rc_calculated
		}
	}

}
