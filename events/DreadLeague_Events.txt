country_event = {
    id = deadempire.30
  	hidden = yes

		is_triggered_only = yes
		immediate = {
			set_country_flag = DED_conversion_on
		}

    option = {
			if = {
				limit = { has_war = no }
				every_owned_state = {
					limit = {
						NOT = { is_core_of = ROOT }
					}
					decrease_population_necromancy = yes
				}
				else = {
					every_state = {
						limit = {
							NOT = { is_core_of = ROOT }
							OR = {
								is_owned_by = ROOT
								is_controlled_by = ROOT
							}
						}
						decrease_population_necromancy = yes
					}
				}
			}
			country_event = { id = deadempire.31 days = 30 }
		}

}
