		allowed = {
			original_tag = CTH
		}

		activation = {
			CVA = {
				exists = no
			}
			OR = {
				controls_state = 1093
				controls_state = 1094
				controls_state = 1095
			}
			NOT = {
				has_country_flag = CTH_supressed_coltva
			}

				has_country_flag = CTH_coltva_revolte

		}

		}

			}

				controls_state = 1095
			}
			NOT = {
				has_country_flag = CTH_supressed_coltva
			}
			NOT = {
				has_country_flag = CTH_coltva_revolte
			}
		}

		is_good = no

		days_mission_timeout = 10

		fire_only_once = yes

		timeout_effect = {
			set_country_flag = CTH_coltva_revolte
			release = CVA
			declare_war_on = {
				target = CVA
				type = annex_everything
			}
			CVA = {
				set_cosmetic_tag = CVA_army
				inherit_technology = ROOT
				load_oob = CVA_rebellion

				limit = {
					1093 = {

					}
				}

				add_state_claim = 1093
			}
			if = {

					1094 = {
						is_core_of = CTH
					}
				}
				remove_state_core = 1094
				add_state_claim = 1094

			if = {

						is_core_of = CTH
					}
				}
				remove_state_core = 1095
				add_state_claim = 1095
			}
		}
	}

	CTH_constitutionalist_rebellion = {
		allowed = {
			original_tag = CTH
		}

		activation = {
			NOT = {
				has_government = democratic
			}
			NOT = {
				has_country_flag = CTH_constitutionalists_suppressed
			}
		}

		available = {
			OR = {
				has_government = democratic
				has_country_flag = CTH_constitutionalists_suppressed
			}
			
		}

		visible = {
			NOT = {
				has_country_flag = CTH_constitutionalists_suppressed
			}
			NOT = {
				has_government = democratic
			}
		}

		is_good = no

		days_mission_timeout = 15

		fire_only_once = yes

		timeout_effect = {
			set_variable = {
				var = civil_war_size_var
				value = party_popularity@democratic
			}
			if = {
				limit = {
					check_variable = {
						var = civil_war_size_var
						value = 0.3
						compare = less_than
					}
				}
				set_variable = {
					var = civil_war_size_var
					value = 0.3
				}
			}
			start_civil_war = {
				ideology = democratic
				size = civil_war_size_var
				capital = 709
				states = {
					876 709 877 699
				}
				keep_unit_leaders_trigger = {
					OR = {
						has_trait = zarcid_supporter
						has_trait = traditionalist_supporter
						has_trait = pirate_supporter
						has_trait = scientific_supporter
					}
				}
			}
			random_other_country = {
				limit = {
					original_tag = CTH
					has_government = democratic
				}
				set_country_flag = CTH_constitutionalists
			}
		}
	}
