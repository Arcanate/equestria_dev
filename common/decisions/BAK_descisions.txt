BAK_elections_category = {
    BAK_election_mission = {
		icon = generic_ignite_civil_war
		available = {
			always = NO
		}
		visible = {
			has_idea = BAK_elections
		}
		activation = {
			has_idea = BAK_elections
		}
		is_good = no
		days_mission_timeout = 335
		fire_only_once = yes
		timeout_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision timeout BAK_election_mission"
			if = { #Fascists Win
				limit = { 
				    AND = {
					    check_variable = { BAK_fascist_support_v > BAK_communist_support_v }
						check_variable = { BAK_fascist_support_v > BAK_neutrality_support_v }
						check_variable = { BAK_fascist_support_v > BAK_harmonist_support_v }
					}
				}
				country_event = { 
				    id = bakara.1
				}
			}
			else_if = { #Communists Win
				limit = { 
				    AND = {
					    check_variable = { BAK_communist_support_v > BAK_fascist_support_v }
						check_variable = { BAK_communist_support_v > BAK_neutrality_support_v }
						check_variable = { BAK_communist_support_v > BAK_harmonist_support_v }
					}
				}
				country_event = { 
				    id = bakara.2
				}
			}
			else_if = { #Neutrality Win
				limit = { 
				    AND = {
					    check_variable = { BAK_neutrality_support_v > BAK_fascist_support_v }
						check_variable = { BAK_neutrality_support_v > BAK_communist_support_v }
						check_variable = { BAK_neutrality_support_v > BAK_harmonist_support_v }
					}
				}
				country_event = { 
				    id = bakara.3
				}
			}
			else = { #Harmonists Win
			    country_event = { 
				    id = bakara.4
				}
			}
			hidden_effect = {
				set_global_flag = JER_revolution_begun
			}
		}
	}
}
