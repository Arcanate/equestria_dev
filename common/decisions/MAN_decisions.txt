MAN_love_tax_category = {

	MAN_love_tax_mission = {

		allowed = { tag = MAN }

		icon = ger_mefo_bills ## make an icon for the love tax ##

        activation = {                         
			always = no
        }
		
		available = {
			has_government = fascism
			is_puppet_of = CHN
		}

		cost = 10

		selectable_mission = yes
		days_mission_timeout = 180
		is_good = no
		cancel_trigger = {
			hidden_trigger = {
				OR = {
					NOT = { is_puppet_of = CHN }
				}
			}
		}
		
		complete_effect = { ## pay up ##
			activate_mission = MAN_paying_the_love_tax
			custom_effect_tooltip = MAN_effects_of_not_paying_the_love_tax
			hidden_effect = {
				remove_ideas = MAN_not_pay_love_tax
				add_ideas = MAN_ahh_i_hate_comiing_up_with_these_names
				CHN = { add_ideas = MAN_lovetax_new_chn_benefits }
			}
		}
		timeout_effect = { ## fail ##
			add_political_power = -150
			custom_effect_tooltip = MAN_autonomy_score_minus_300
			hidden_effect = {
				add_autonomy_score = {
					value = -300 
					localization = MAN_failure_to_meet_love_tax_quotas
				}
			}
			activate_mission = MAN_love_tax_mission
		}

		ai_will_do = {
			factor = 200
		}
	}
	
	MAN_paying_the_love_tax = {

		allowed = { tag = MAN }

		icon = ger_mefo_bills ## make an icon for the love tax ##

        activation = {                                                 
			always = no                 
        }
		
		available = {
			always = no
		}
		
		days_mission_timeout = 180
		## modding notes: i didn't know that if you removed "is_good = no" the wording of the decision would become totally different (and just what I desired) and spent some time trying to find a workaround ... ##
		cancel_trigger = {
			hidden_trigger = {
				OR = {
					NOT = { is_puppet_of = CHN }
				}
			}
		}
		
		timeout_effect = { ## the cycle starts anew ##
			custom_effect_tooltip = MAN_autonomy_score_add_50
			hidden_effect = {
				add_autonomy_score = {
					value = 50 
					localization = MAN_met_love_tax_quotas
				}
			}
			activate_mission = MAN_love_tax_mission
			custom_effect_tooltip = MAN_in_the_process_of_paying_the_love_tax
			hidden_effect = {
				add_ideas = MAN_not_pay_love_tax
				remove_ideas = MAN_ahh_i_hate_comiing_up_with_these_names
				CHN = { remove_ideas = MAN_lovetax_new_chn_benefits }
			}
		}
	}
