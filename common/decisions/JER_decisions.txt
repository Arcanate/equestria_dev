			    country_event = { id = aquileia.10 days = 1 }
			        }
			    }
			}
			if = { 
			limit = { original_tag = JRR }
			JRR = {
			    if = {
			    limit = { check_variable = { monarchist_support > revolutionary_support } }
			    country_event = { id = aquileia.11 days = 1 }
			    }
			    else = {
			    country_event = { id = aquileia.12 days = 1 }
			        }
			    }
			}
			if = { 
			limit = { original_tag = VIN }
			VIN = {
			    if = {
			    limit = { check_variable = { monarchist_support > revolutionary_support } }
			    country_event = { id = aquileia.13 days = 1 }
			    }
			    else = {
			    country_event = { id = aquileia.14 days = 1 }
			        }
			    }
			}
			if = { 
			limit = { original_tag = JFT }
			JFT = {
			    if = {
			    limit = { check_variable = { monarchist_support > revolutionary_support } }
			    country_event = { id = aquileia.15 days = 1 }
			    }
			    else = {
			    country_event = { id = aquileia.45 days = 1 }
			        }
			    }
			}
			if = {
			limit = { original_tag = JER }
			JER = {
			    country_event = { id = aquileia.46 days = 4 }
			    }
			}

			    set_global_flag = JER_revolution_begun
			}
        }
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision timeout JER_revolution_mission"
			if = {
			limit = { original_tag = JER }
			JER = {
			    country_event = { id = aquileia.47 days = 1 }
