JER_parliament_category = {
	# activate with activate_JER_parliament_lose_support_mission = yes
	JER_parliament_lose_support_mission = {
		icon = ger_mefo_bills
		allowed = {
			always = no
		}
		is_good = no
		selectable_mission = yes
		days_mission_timeout = 60
		cost = 50
		timeout_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision timeout JER_parliament_lose_support_mission"
			if = {
				limit = {
					check_variable = {
						JER_parliament_lose_support_mission_party = token:communism
					}
				}
				subtract_from_variable = { 
					var = JER.communist_support 
					value = 10 
				}
				clamp_variable = {
					var = JER.communist_support 
					min = 0 
					max = JER.communist_max
				}
			}
			else_if = {
				limit = {
					check_variable = {
						JER_parliament_lose_support_mission_party = token:democratic
					}
				}
				subtract_from_variable = { 
					var = JER.democratic_support 
					value = 10 
				}
				clamp_variable = {
					var = JER.democratic_support 
					min = 0 
					max = JER.democratic_max
				}
			}
			else_if = {
				limit = {
					check_variable = {
						JER_parliament_lose_support_mission_party = token:neutrality
					}
				}
				subtract_from_variable = { 
					var = JER.neutrality_support 
					value = 10 
				}
				clamp_variable = {
					var = JER.neutrality_support 
					min = 0 
					max = JER.neutrality_max
				}
			}
			else = {
				subtract_from_variable = { 
					var = JER.fascism_support 
					value = 10 
				}
				clamp_variable = {
					var = JER.fascism_support 
					min = 0 
					max = JER.fascism_max
				}
			}
			hidden_effect = {
				update_JER_parliament_max = yes
				activate_JER_parliament_lose_support_mission = yes
			}
		}
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision JER_parliament_lose_support_mission"
		}
	}
	JER_parliament_decision_1 = {
		icon = ger_mefo_bills
		available = {
			always = no
		}
		visible = {
			has_completed_focus = JER_the_triumph
			is_subject = no
		}
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision JER_parliament_decision_1"
		}
	}
}
