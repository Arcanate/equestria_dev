ANG_secret_maar = {
	ANG_hide_cult = {
		is_good = NO
		priority = 10		
		activation  = {
		    has_country_flag = ANG_secret_maarite
			country_exists = event_target:griffonian_empire
			NOT = { 
				has_war_with = event_target:griffonian_empire
			}
		}
		cancel_trigger = {
			ROOT = {
				OR = {
				    has_country_flag = ANG_revealed_maarite
					tag = event_target:griffonian_empire
					has_civil_war = yes
					has_war_with = event_target:griffonian_empire
				}
			}
		}
		cost = ANG_hiding_cost_total?5
		selectable_mission = yes
		fire_only_once = no
		
		days_remove = ANG_hide_days
		complete_effect = {
			custom_effect_tooltip = ANG_hide_cult_days_tp
			if = {
				limit = {
					check_variable = { var = ANG_hiding_cost value = 15 compare = less_than }
				}
				set_variable = { var = ANG_hiding_cost value = 10 }
			}
			add_to_variable = { var = ANG_hiding_cost value = 10}
			set_variable = { ANG_hiding_cost_total = ANG_hiding_cost }
			
			set_variable = { var = ANG_hiding_cost_states value = num_controlled_states }			
			multiply_variable = { ANG_hiding_cost_states = 3 }			
			add_to_variable = { var = ANG_hiding_cost_total value = ANG_hiding_cost_states}
			
			set_variable = { ANG_hide_days = days_mission_timeout@ANG_hide_activity }
			add_to_variable = { var = ANG_hide_days value = 60 }
		}
		days_mission_timeout = 30
		timeout_effect = {
			# event_target:griffonian_empire = { country_event = { days = 1 id = angriver.92 } }
		}
		ai_will_do = {
			factor = 200
		}
	}
}
