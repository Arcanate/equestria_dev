on_actions = {

  #Not run on a specific country.
  on_startup = {
    effect = {
      every_country = {
        limit = { is_generic_race = yes }
        set_technology = { generic_race = 1 }
      }

    }
  }

  #used when puppeting in a peace conference
	#ROOT = nation being puppeted, FROM = overlord
  on_puppet = {
    effect = {
    if = {
      limit = {
        FROM = {
          is_ai = no
          is_eqs_civil_war_tag = yes
         }
        ROOT = {
          is_ai = no
          is_eqs_civil_war_tag = yes
        }
      }
      FROM = { annex_country = ROOT }
    }
      if = {
        limit = {
          EQS = { is_ai = yes }
          original_tag = CHN
          NOT = { country_exists = OLE }
          FROM = {
            original_tag = EQS
            has_government = democratic
          }
          any_owned_state = {
            is_core_of = OLE
          }
        }
        every_owned_state = {
          limit = { is_core_of = OLE }
          OLE = { transfer_state = PREV }
        }
        OLE = {
          set_politics = { ruling_party = democratic elections_allowed = no }
          set_political_party = { ideology = democratic popularity = 1.0 }
        }
        EQS = {
          puppet = OLE
          transfer_state = 216
          transfer_state = 217
          transfer_state = 219
          transfer_state = 218
          transfer_state = 69
          transfer_state = 224
          transfer_state = 296
          transfer_state = 246
          transfer_state = 291
          transfer_state = 248
          transfer_state = 316
          transfer_state = 319
          transfer_state = 247
        }
      }
    }
  }

  #ROOT is winner #FROM gets annexed - For civil wars on_civil_war_end is also fired
	on_annex = {
		effect = {
			if = {
        limit = {
          original_tag = OLE
          NOT = { tag = OLE }
          FROM = { original_tag = OLE }
        }
        set_cosmetic_tag = OLE_CWWON
      }
		}
	}

  on_civil_war_end = {
		effect = {
			if = {
        limit = {
          original_tag = OLE
          is_in_faction_with = EQS
          EQS = { has_country_flag = EQS_OLE_temp_faction }
        }
        EQS = {
          remove_from_faction = ROOT
          dismantle_faction = yes
          clr_country_flag = EQS_OLE_temp_faction
        }
      }
		}
	}
}
