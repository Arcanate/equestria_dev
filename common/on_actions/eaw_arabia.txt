on_actions = {
    on_weekly_arabia = {
	    effect = {
        	if = {
	        	limit = {
			        has_idea = arabian_civil_war
		    	}
			    add_to_variable = {
		    	    var = war_exhaustion
				    value = 1
			    }
			    clamp_variable = {
            	    var = war_exhaustion
           	 	    min = 1
            	    max = 100
        	    }
		    }
        }
    }
	on_daily = {
	    effect = {
        	if = {
	        	limit = {
			        has_idea = arabian_civil_war
		    	}
                calculate_arabian_civil_war_stability = yes
				calculate_arabian_civil_war_war_support = yes
				calculate_arabian_civil_war_conscription_factor = yes
				calculate_arabian_civil_war_surrender_limit = yes
		    }
        }
	}
}
