	name = "Ignore Excavation Categories"
	desc = ""
	
	enable = {
		check_variable = { modifier@min_export = 0 }
		
		if = {
			limit = {
				check_variable = { resource_produced@steel > 0 }
			}
			check_variable = { resource_imported@steel = 0 }
		}
		if = {
			limit = {
				check_variable = { resource_produced@tungsten > 0 }
			}
			check_variable = { resource_imported@tungsten = 0 }
		}
		if = {
			limit = {
				check_variable = { resource_produced@aluminium > 0 }
			}
			check_variable = { resource_imported@aluminium = 0 }
		}
		if = {
			limit = {
				check_variable = { resource_produced@rubber > 0 }
			}
			check_variable = { resource_imported@rubber = 0 }
		}
		if = {
			limit = {
				check_variable = { resource_produced@chromium > 0 }
			}
			check_variable = { resource_imported@chromium = 0 }
		}
		if = {
			limit = {
				check_variable = { resource_produced@crystals > 0 }
			}
			check_variable = { resource_imported@crystals = 0 }
		}
		if = {
			limit = {
				fuel_ratio < 0.25
			}
			any_owned_state = { has_resources_amount = { resource = oil amount > 0} }
		}
	}
	abort = {
		OR = {
			check_variable = { modifier@min_export > 0 }
			
			AND = {
				check_variable = { resource_produced@steel > 0 }
				check_variable = { resource_imported@steel > 0 }
			}
			AND = {
				check_variable = { resource_produced@tungsten > 0 }
				check_variable = { resource_imported@tungsten > 0 }
			}
			AND = {
				check_variable = { resource_produced@aluminium > 0 }
				check_variable = { resource_imported@aluminium > 0 }
			}
			AND = {
				check_variable = { resource_produced@rubber > 0 }
				check_variable = { resource_imported@rubber > 0 }
			}
			AND = {
				check_variable = { resource_produced@chromium > 0 }
				check_variable = { resource_imported@chromium > 0 }
			}
			AND = {
				check_variable = { resource_produced@crystals > 0 }
				check_variable = { resource_imported@crystals > 0 }
			}
			AND = {
				fuel_ratio > 0.5
				any_owned_state = { has_resources_amount = { resource = oil amount > 0 } }
			}
		}
	}

	weight = {
		base = 1
	}
