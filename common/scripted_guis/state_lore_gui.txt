scripted_gui = {
	state_lore_container_gui = {
		context_type = selected_state_context
		parent_window_token = selected_state_view
		
		window_name = "state_lore_container"
		
		visible = {
			THIS = { 
				state = 75 ## choose the state carefully so the button doesn't cover up the state name display ##
			}
		}
		
		effects = {
			state_lore_button_click = {
				if = {
					limit = { THIS = { state = 75 } }
					ROOT = { set_variable = { state_lore_text_container_variable = 75 } }
				}
			}
		}
	}
	
	state_lore_container_text_gui = {
		context_type = player_context
		
		window_name = "state_lore_text_container"
		
		visible = {
			ROOT = { 
				check_variable = { state_lore_text_container_variable > 0 }
			}
		}
		
		effects = {
			state_lore_text_close_button_click = {
				ROOT = { 
					set_variable = { state_lore_text_container_variable = 0 }
				}
			}
		}
	}
}
