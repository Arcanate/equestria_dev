#For EQS civil war

eqs_civilwar_peace = {
	# ROOT = Taker, FROM = Giver
	enable = {
    AND = {
			OR = {
				is_eqs_civil_war_tag = yes
				any_country = {
					AND = {
						is_in_peace_conference = yes
						is_eqs_civil_war_tag = yes
						OR = {
							is_in_faction_with = ROOT
							has_war_together_with = ROOT
						}
					}
				}
				has_global_flag = equestrian_civil_war
				has_global_flag = equestrian_com_civil_war
			}
      FROM = {
        is_eqs_civil_war_tag = yes
      }
    }
	}

	annex_randomness = 0
	liberate_randomness = 0
	puppet_randomness = 0
	take_states_randomness = 0
	force_government_randomness = 0

	# ROOT = Taker, FROM = Giver
	annex = {
		factor = 10
		modifier = {
			factor = 0
			AND = {
				ROOT = { is_eqs_civil_war_tag = no }
				FROM = { is_eqs_civil_war_tag = no }
			}
		}
	}

	# ROOT = Taker, FROM = Liberated
	liberate = {
		factor = 0
	}

	# ROOT = Taker, FROM = Giver
	puppet = {
		factor = 0
	}

	# ROOT = Taker, FROM = State
	take_states = {
		factor = 10000
		modifier = {
			factor = 0
			FROM = {
				owner = { is_eqs_civil_war_tag = no }
			}
		}
	}

	# ROOT = Taker, FROM = Giver
	force_government = {
		factor = 0
	}
}
