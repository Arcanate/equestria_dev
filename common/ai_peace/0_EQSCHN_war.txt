# Default Logic behaviour that the AI will fallback on if they are in a civil war

eqs_chn_peace = {
	# ROOT = Taker, FROM = Giver
	enable = {
    AND = {
      original_tag = EQS
			has_government = democracy
			from = {
					original_tag = CHN
			}
    }
	}

	annex_randomness = 0
	liberate_randomness = 0
	puppet_randomness = 0
	take_states_randomness = 0
	force_government_randomness = 0

	# ROOT = Taker, FROM = Giver
	annex = {
		factor = 0
	}

	# ROOT = Taker, FROM = Liberated
	liberate = {
		factor = 6000
		modifier = {
				factor = 0
				NOT = { original_tag = OLE }
			}
		}
	}

	# ROOT = Taker, FROM = Giver
	puppet = {
		factor = 4000


		modifier = {
				factor = 0
				any_owned_state = {
				OR = {
					is_claimed_by = ROOT
					is_core_of = ROOT
				}
			}
		}
	}

	# ROOT = Taker, FROM = State
	take_states = {
		factor = 4000

		modifier = {
			factor = 0

			FROM = {
				NOT = {
					is_claimed_by = ROOT
					is_core_of = ROOT
				}
			}
		}
	}

	# ROOT = Taker, FROM = Giver
	force_government = {
		factor = 0
	}
}
