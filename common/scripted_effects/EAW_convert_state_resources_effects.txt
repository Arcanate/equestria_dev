
turn_oil_into_aluminium_state = {
    add_resource = {
        type = aluminium
        amount = resource@oil
    }
    set_temp_variable = { neg_res = resource@oil }
    multiply_temp_variable = { neg_res = -1 }
    add_resource = {
        type = aluminium
        amount = neg_res
    }
