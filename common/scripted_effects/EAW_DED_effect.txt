DED_startup = {
	set_technology = {
		DED_undeads = 1
		DED_stitchers_creation_tech = 1
		popup = no
	}
	load_oob = "DED_1007"
	add_equipment_to_stockpile = {
		type = undead_equipment_0
    	amount = 900
	}


	if = {
		limit = {
			has_dlc = "No Step Back"
		}
		create_equipment_variant = {
			name = "Abomination"
			type = DED_stitcher_creation_1
			parent_version = 0
			modules = {
				main_armament_slot = DED_tank_heavy_machine_gun
			}
			is_buildable = no
			icon = "gfx/interface/technologies/gwtank.dds"
		}
		create_equipment_variant = {
			name = "Abomination 2.0"
			type = DED_stitcher_creation_1
			parent_version = 0
			is_buildable = no
			modules = {
				main_armament_slot = DED_tank_heavy_machine_gun
			}
			icon = "gfx/interface/technologies/gwtank.dds"
		}

		#To add tanks, you need to precise: Type, Variant AND Producer. Not specifying the producer will only add the latest version of the type.
		add_equipment_to_stockpile = {
			type = DED_stitcher_creation_1
	    	amount = 900
	    	producer = DED
	    	variant_name = "Abomination"
		}
		add_equipment_to_stockpile = {
			type = DED_stitcher_creation_1
	    	amount = 900
	    	producer = DED
	    	variant_name = "Abomination 2.0"
		}
	}

	load_focus_tree = DED_START
	#army_experience = 100
	#set_variable = { DED_undead_production = 735 }

	country_event = { id = dread.1 days = 3 }
}

DED_monthly = {
	if = {
		limit = {
			has_idea = DED_undead_production_idea
		}
		set_temp_variable = {
			DED_total = max_manpower_k
		}
		divide_temp_variable = {
			DED_total = 12
		}
		multiply_temp_variable = {
			DED_total = 5
		}
		round_temp_variable = DED_total
		set_variable = {
			DED_undead_production = DED_total
		}
		set_temp_variable = {
			DED_undead_equipement_gain = DED_undead_production
		}
		divide_temp_variable = {
			DED_undead_equipement_gain = 100
		}
		round_temp_variable = DED_undead_equipement_gain
		add_equipment_to_stockpile = {
			type = undead_equipment_0
			amount = DED_undead_equipement_gain
		}
	}
}
