decrease_partisans = {
	if = {
		limit = {
			has_idea = MGY_post_cw_paramilitaries
		}
		swap_ideas = {
			remove_idea = MGY_post_cw_paramilitaries
			add_idea = MGY_post_cw_paramilitaries_2
		}
	}
	else_if = {
		limit = {
			has_idea = MGY_post_cw_paramilitaries_2
		}
		swap_ideas = {
			remove_idea = MGY_post_cw_paramilitaries_2
			add_idea = MGY_post_cw_paramilitaries_3
		}
	}
	else_if = {
		limit = {
			has_idea = MGY_post_cw_paramilitaries_3
		}
		swap_ideas = {
			remove_idea = MGY_post_cw_paramilitaries_3
			add_idea = MGY_post_cw_paramilitaries_4
		}
	}
	else_if = {
		limit = {
			has_idea = MGY_post_cw_paramilitaries_4
		}
		remove_ideas = MGY_post_cw_paramilitaries_4
	}	
}
