can_core_nation_through_compliance = {
	OR = {
		any_of_scopes = {
			array = potential_cores
			original_tag = FROM
		}
		if = {
			limit = {
				check_variable = {
					num_controlled_states < FROM.core_states^num
				}
			}
			any_controlled_state = {
				occupied_country_tag = FROM
				OR = {
					any_of = {
						array = core_countries
						is_in_array = {
							ROOT.potential_cores = var:v
						}
					}
					any_of = {
						array = state_original_cores
						is_in_array = {
							ROOT.potential_original_cores = var:v
						}
					}
				}
			}
		}
		else = {
			any_of_scopes = {
				array = FROM.core_states
				is_controlled_by = ROOT
				OR = {
					any_of = {
						array = core_countries
						is_in_array = {
							ROOT.potential_cores = var:v
						}
					}
					any_of = {
						array = state_original_cores
						is_in_array = {
							ROOT.potential_original_cores = var:v
						}
					}
				}
			}
		}
	}
}
