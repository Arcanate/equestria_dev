can_core_nation_through_compliance = {
	OR = {
		is_in_array = {
			potential_cores = FROM
		}
		AND = {
			is_in_array = {
				potential_original_cores = FROM
			}
			any_of_scopes = {
				array = FROM.original_cores 
				is_owned_by = ROOT
			}
		}
	}
}

can_core_nation_through_compliance_state = {
	OR = {
		is_in_array = {
			FROM.potential_cores = FROM.FROM
		}
		AND = {
			is_in_array = {
				FROM.potential_original_cores = FROM.FROM
			}
			is_in_array = {
				FROM.FROM.original_cores = THIS
			}
		}
	}
}
