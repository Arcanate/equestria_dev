can_core_nation_through_compliance = {
	OR = {
		any_of_scopes = {
			array = potential_cores
			original_tag = FROM
		}
		AND = {
			any_of_scopes = {
				array = potential_original_cores
				original_tag = FROM
			}
			any_of_scopes = {
				array = FROM.original_cores 
				is_owned_by = ROOT
			}
		}
	}
}

can_core_nation_through_compliance_state = {
	OR = {
		any_of_scopes = {
			array = FROM.potential_cores
			original_tag = FROM.FROM
		}
		AND = {
			any_of_scopes = {
				array = FROM.potential_original_cores
				original_tag = FROM.FROM
			}
			is_in_array = {
				FROM.FROM.original_cores = THIS
			}
		}
	}
}

can_core_nation_through_compliance_state_ROOT = {
	OR = {
		any_of = {
			array = state_original_cores
			is_in_array = {
				ROOT.potential_original_cores = var:v
			}
		}
		any_of = {
			array = global.countries
			is_core_of = var:v
			is_in_array = {
				ROOT.potential_cores = var:v
			}
		}
	}
}

can_core_nation_through_compliance_state_PREV = {
	OR = {
		any_of = {
			array = state_original_cores
			is_in_array = {
				PREV.potential_original_cores = var:v
			}
		}
		any_of = {
			array = global.countries
			is_core_of = var:v
			is_in_array = {
				PREV.potential_cores = var:v
			}
		}
	}
}
