compliance_40_core = {
	type = core_compliance_modifier
	icon = "GFX_occupation_compliance_modifier_strip:5"
	small_icon = "GFX_occupation_compliance_modifier_small_strip:5"
	threshold = 40
	margin = 2
	state_modifier = {
	}
	visible = {
		can_core_nation_through_compliance = yes
	}
	
	on_enable = {
		if = {
			limit = {
				is_in_array = {
					potential_cores = FROM
				}
			}
			every_owned_state = {
				limit = { is_core_of = FROM }
				add_core_of = ROOT
			}
		}
		else = {
			every_owned_state = {
				limit = {
					is_core_of = FROM
					is_in_array = {
						FROM.original_cores = THIS
					}
				}
				add_core_of = ROOT
			}
		}
	}
}
