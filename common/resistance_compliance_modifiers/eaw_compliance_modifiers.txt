compliance_40_core = {
	type = core_compliance_modifier
	icon = "GFX_occupation_compliance_modifier_strip:5"
	small_icon = "GFX_occupation_compliance_modifier_small_strip:5"
	threshold = 40
	margin = 2
	state_modifier = {
	}
	visible = {
		can_core_nation_through_compliance = yes
	}
	
	on_enable = {
		if = {
			limit = {
				any_of_scopes = {
					array = potential_cores
					original_tag = FROM
				}
			}
			every_owned_state = {
				limit = { is_core_of = FROM }
				add_core_of = ROOT
			}
		}
		else = {
			every_owned_state = {
				limit = {
					is_core_of = FROM
					OR = {
						any_of = {
							array = state_original_cores
							is_in_array = {
								ROOT.potential_original_cores = var:v
							}
						}
						any_of = {
							array = ROOT.potential_cores
							is_core_of = var:v
						}
					}
				}
				add_core_of = ROOT
			}
		}
	}
}
