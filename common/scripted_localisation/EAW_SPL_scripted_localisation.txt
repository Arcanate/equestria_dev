# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes


defined_text = {
	name = Stagpal_17_Description

	text = {
		trigger = {
			AND = {
				has_country_flag = SPL_old_deers
				has_country_flag = SPL_old_llamas
			}
		}		
		localization_key = SPL_17_Old_Old
	}
	text = {
		trigger = {	
			AND = {
				has_country_flag = SPL_old_deers
				has_country_flag = SPL_new_llamas
			}
		}		
		localization_key = SPL_17_Old_New
	}
	text = {
		trigger = {
			AND = {
				has_country_flag = SPL_new_deers
				has_country_flag = SPL_old_llamas
			}	
		}	
		localization_key = SPL_17_New_Old
	}
	text = {
		trigger = {
			AND = {
				has_country_flag = SPL_new_deers
				has_country_flag = SPL_new_llamas
			}
		}		
		localization_key = SPL_17_New_New
	}
}

defined_text = {
    name = SPL_Council_Status

    text = {
        trigger = {
            set_temp_variable = { llamas_power_plus_1 = llamas_power }
            add_to_temp_variable = { llamas_power_plus_1 = 1 }
            check_variable = { 
                deer_power > llamas_power_plus_1
            }
        }
        localization_key = SPL_Council_Deer
    }
    text = {
        trigger = {
            set_temp_variable = { deer_power_plus_1 = deer_power }
            add_to_temp_variable = { deer_power_plus_1 = 1 }
            check_variable = { 
                llamas_power > deer_power_plus_1 
            }
        }
        localization_key = SPL_Council_Llama
    }
    text = {
        localization_key = SPL_Council_good
    }
}
