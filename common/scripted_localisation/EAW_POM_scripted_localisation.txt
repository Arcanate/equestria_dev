# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = { # POM_A_Republic_of_Harmony focus name
	name = GetPOM_A_Republic_of_Harmony_Name
	text = {
		trigger = {
		   has_government = neutrality #NOTE TO SELF: SWITCH THIS TO A COUNTRY FLAG
		}
		localization_key = POM_A_Republic_of_Harmony_Name_Broken
	}
	text = {
		trigger = {
			has_government = fascism #NOTE TO SELF: SWITCH THIS TO A COUNTRY FLAG
		}
		localization_key = POM_A_Republic_of_Harmony_Name_Fascist
	}
	text = {
		trigger = {
			has_government = democratic #NOTE TO SELF: SWITCH THIS TO A COUNTRY FLAG
		}
		localization_key = POM_A_Republic_of_Harmony_Name_Harmony
	}
  text = {
		trigger = {
		   has_government = communism #NOTE TO SELF: SWITCH THIS TO A COUNTRY FLAG
		}
		localization_key = POM_A_Republic_of_Harmony_Name_Communist
	}
}

defined_text = { # POM_Lu_Republikan_Broadel focus name
	name = GetPOM_Lu_Republikan_Broadel_Name
	text = {
		trigger = {
		   has_government = democratic
		}
		localization_key = POM_Lu_Republikan_Broadel_Harmony
	}
	text = {
		trigger = {

				has_government = democratic #NOTE TO SELF: SWITCH THIS TO A COUNTRY FLAG
			}
		}
		localization_key = POM_Lu_Republikan_Broadel_Fascist
	}
}
